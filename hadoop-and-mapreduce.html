ggplot(result1,aes(x=hour))+
  scale_x_continuous(limits=c(0,24),
                     breaks=seq(0,24,2))+
  geom_histogram(aes(fill = ..count..),binwidth =0.5)+
  xlab('Hour')+
  ylab('The Frequency of Post by Students')+
  ggtitle('The Post Frequency in Each Hour')
